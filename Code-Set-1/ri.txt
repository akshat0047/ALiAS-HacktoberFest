#include<stdio.h>
#include<string.h>
int main()
{
int i,j,sum1=0,sum2=0,l;
int x[3][3]={{1,2,3},
{4,5,6},
{17,8,9}
};
for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
if(i==j)
{sum1=sum1+x[i][j];
}
}
for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
{
if((i+j)==2)
{
    sum2=sum2+x[i][j];
}
}
}
l=sum1-sum2;
if(l<0)
{
    l=l*(-1);
}
printf("difference between the diagonal is %d",l);
getch();
}
